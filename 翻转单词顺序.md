### 翻转单词顺序
输入一个英文句子，翻转句子中单词的顺序，但单词内字符的顺序不变，标点符号和普通字符一样处理。
```cpp
//依次将每个单词插入到字符串的首部
string ReverseSentence(string str) 
{
    string result;
    if(str.empty())    return result;
    int i = 0, j = 0;
    int len = str.length();
    while(j < len)
    {
        if(str[j] == ' ')   //i为单词的首字符，j为单词结束后的空格（对于连续多个空格的情况也能有效处理）
        {
            string tmp = " " + str.substr(i, j - i);
            result = tmp + result;
            i = j + 1;
        }  
        j++;
    }
    result = str.substr(i) + result;
    return result;
}

//翻转整个句子，再翻转每个单词
string ReverseSentence(string str) 
{
    int i = 0, start = 0, end = 0;
    int len = str.length();
    reverse(str, 0, len - 1);
    while(i < len)
    {
        while(i < len && str[i] == ' ')
            i++;
        start = i;
        while(i < len && str[i] != ' ')
            i++;
        end = i;
        reverse(str, start, end - 1);
    }  
    return str;
}

void reverse(string& word, int start, int end)
{
    while(start < end)
        swap(word[start++], word[end--]);
}
```
